```typescript
      'chatbot',           // כי בחר sales
```

## 💻 קוד פסאודו - הזרימה המלאה

```typescript
// === 1. משתמש ממלא טופס ===
const formData = {
  ai_agents: { count: '2', channels: ['whatsapp'], domains: ['sales'] },
  automations: { processes: ['lead_management'], time_wasted: '3-4h', ... },
  crm: { exists: 'yes', system: 'zoho', ... }
};

// === 2. המרה לפורמט מלא ===
```
